resource "aws_s3_bucket" "resume" {
  bucket = var.bucket
 

  tags = {
    Name        = "My bucket"
    Environment = "Dev"
  }
}





  resource "aws_s3_object" "index" {
bucket = var.bucket

  source = "index.html"
   content_type = "text/html"

  aws_s3_bucket.resume
 ]

}
